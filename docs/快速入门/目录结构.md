# 目录结构

```shell script
├── app                  [App 目录]
│   ├── Command              [命令目录]
│   ├── Controller           [控制器目录]
│   ├── Event                [事件目录]
│   ├── Listener             [监听目录]
│   ├── Middleware           [中间件目录]
│   ├── Model                [模型目录]
│   └── Provider             [服务目录]
├── config               [配置目录]
│   ├── app.php              [App 配置]
│   ├── console.php          [命令行配置]
│   ├── database.php         [数据库配置]
│   ├── event.php            [事件配置]
│   ├── logger.php           [日志配置]
│   └── view.php             [视图配置]
├── public               [Web 根目录]
│   ├── index.php            [Web 入口文件]
│   └── static               [静态资源目录]
├── route                [路由目录]
│   ├── api.php              [Api 路由文件]
│   └── web.php              [Web 路由文件]
├── runtime              [运行时目录]
│   ├── logs                 [日志问价目录]
│   └── views                [视图缓存文件目录]
├── tests                [测试目录]
├── vendor               [Vendor 目录]
├── view                 [视图目录]
├── .env                 [Env 文件]
├── .env.example         [Env 示例文件]
├── coole                [命令行入口文件]
├── CHANGELOG.md         [版本变更日志文件]
├── LICENSE              [协议文件]
└── README.md            [Readme 文件]
```
